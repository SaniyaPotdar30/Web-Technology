ng if-else, for loop, switch cases- how to use in angular

<!-- app.component.html -->

<!-- IF-ELSE using ngIf -->
<p *ngIf="isLoggedIn; else loginMsg">Welcome User!</p>
<ng-template #loginMsg>
  <p>Please login</p>
</ng-template>

<hr>

<!-- FOR LOOP using ngFor -->
<ul>
  <li *ngFor="let item of items">
    {{ item }}
  </li>
</ul>

<hr>

<!-- SWITCH CASE using ngSwitch -->
<div [ngSwitch]="role">
  <p *ngSwitchCase="'admin'">Admin Panel</p>
  <p *ngSwitchCase="'user'">User Dashboard</p>
  <p *ngSwitchDefault>Guest View</p>
</div>


// app.component.ts

export class AppComponent {
  isLoggedIn = true;

  items = ['Apple', 'Mango', 'Banana'];

  role = 'admin';
}
